Next-Message Star

A lightweight feedback message appearing at the top of the screen, fading automatically. Has many different options for different situations.

Doc

Install

$ yarn add next-message

Use

import $message from 'next-message';
import 'next-message/index.css';

Example

$message.info('info message')
$message.success('success message')
$message.warning('warning message')
$message.error('error message')
$message.loading('loading...')

$message.destroy()

$message.info('This is a message',{
      key: 'msg-update',
      onClose:() => console.log('close'),
      closable: true,
      duration: 10
});
setTimeout(() => {
  $message.update({
    key: 'msg-update', 
    content: 'This is a updated message'
  });
}, 1500);



$message.info({
    content: 'string message'
})

$message.info('This is a message with a close button',{
    onClose:() => console.log('close'),
    closable: true,
    duration: 2
})

$message.info('This is a message with a manual close button',{
    onClose:() => console.log('close'),
    closable: true,
    duration: 0
})

// default
$message.info('This a message <span style="color:red">red color</span>',{
    duration: 0,
    closable: true
})
